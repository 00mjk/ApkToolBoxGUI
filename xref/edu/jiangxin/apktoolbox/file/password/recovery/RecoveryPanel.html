248     private void onStart() {
249         FileChecker fileChecker = (FileChecker) checkerTypeComboBox.getSelectedItem();
250         if (fileChecker == null) {
251             JOptionPane.showMessageDialog(this, "fileChecker is null");
252             return;
253         }
254         if (!fileChecker.prepareChecker()) {
255             JOptionPane.showMessageDialog(this, "onStart failed: Checker condition does not been prepared!");
256             return;
257         }
258 
259         File selectedFile = recoveryFilePanel.getFile();
260         if (!selectedFile.isFile()) {
261             JOptionPane.showMessageDialog(this, "file is null");
262             return;
263         }
264 
265         String extension = FilenameUtils.getExtension(recoveryFilePanel.getFile().getName());
266         if (!ArrayUtils.contains(fileChecker.getFileExtensions(), StringUtils.toRootLowerCase(extension))) {
267             JOptionPane.showMessageDialog(this, "invalid file");
268             return;
269         }
270 
271         fileChecker.attachFile(selectedFile);
272         currentFileChecker = fileChecker;
273 
274         Component selectedPanel = categoryTabbedPane.getSelectedComponent();
275         if (selectedPanel.equals(bruteForceCategoryPanel)) {
276             currentCategory = CATEGORY.BRUTE_FORCE;
277         } else if (selectedPanel.equals(dictionaryCategoryPanel)) {
278             int threadNum = (Integer) threadNumSpinner.getValue();
279             if (threadNum == 1) {
280                 currentCategory = CATEGORY.DICTIONARY_SINGLE_THREAD;
281             } else {
282                 currentCategory = CATEGORY.DICTIONARY_MULTI_THREAD;
283             }
284         } else {
285             currentCategory = CATEGORY.UNKNOWN;
286         }
287         switch (currentCategory) {
288             case BRUTE_FORCE -> onStartByBruteForceCategory();
289             case DICTIONARY_SINGLE_THREAD -> onStartByDictionarySingleThreadCategory();
290             case DICTIONARY_MULTI_THREAD -> onStartByDictionaryMultiThreadCategory();
291             case UNKNOWN -> JOptionPane.showMessageDialog(this, "onStart failed: Invalid category!");
292         }
293     }

295     private void onStop() {
296         switch (currentCategory) {
297             case BRUTE_FORCE -> onStopByBruteForceCategory();
298             case DICTIONARY_SINGLE_THREAD -> onStopByDictionarySingleThreadCategory();
299             case DICTIONARY_MULTI_THREAD -> onStopByDictionaryMultiThreadCategory();
300             case UNKNOWN -> JOptionPane.showMessageDialog(this, "onStop failed: Invalid category!");
301         }
302     }

304     private void onStartByBruteForceCategory() {
305         logger.info("onStartByBruteForceCategory");
306         StringBuilder sb = new StringBuilder();
307         if (numberCheckBox.isSelected()) {
308             sb.append("0123456789");
309         }
310         if (lowercaseLetterCheckBox.isSelected()) {
311             sb.append("abcdefghijklmnopqrstuvwxyz");
312         }
313         if (uppercaseLetterCheckBox.isSelected()) {
314             sb.append("ABCDEFGHIJKLMNOPQRSTUVWXYZ");
315         }
316         if (sb.length() <= 0) {
317             JOptionPane.showMessageDialog(this, "Character set is empty!");
318             return;
319         }
320         final String charSet = sb.toString();
321 
322         int minLength = (Integer) minSpinner.getValue();
323         int maxLength = (Integer) maxSpinner.getValue();
324         if (minLength > maxLength) {
325             JOptionPane.showMessageDialog(this, "Minimum length is bigger than maximum length!");
326             return;
327         }
328 
329         setIsRecovering(true);
330         String password = null;
331         for (int length = minLength; length <= maxLength; length++) {
332             setProgressMaxValue((int) Math.pow(charSet.length(), length));
333             setProgressBarValue(0);
334             long startTime = System.currentTimeMillis();
335             int numThreads = Math.min(getThreadCount(charSet.length(), length), currentFileChecker.getMaxThreadNum());
336             logger.info("[" + currentFileChecker + "]Current attempt length: " + length + ", thread number: " + numThreads);
337 
338             workerPool = Executors.newFixedThreadPool(numThreads);
339             BruteForceFuture bruteForceFuture = new BruteForceFuture(numThreads);
340             BruteForceTaskConst consts = new BruteForceTaskConst(numThreads, length, currentFileChecker, charSet);
341 
342             for (int taskId = 0; taskId < numThreads; taskId++) {
343                 BruteForceTask task = new BruteForceTask(taskId, true, consts, bruteForceFuture, this);
344                 workerPool.execute(task);
345             }
346             try {
347                 password = bruteForceFuture.get();
348             } catch (Exception e) {
349                 logger.info("Exception test: ", e);
350             } finally {
351                 if (workerPool.isShutdown()) {
352                     workerPool.shutdown();
353                 }
354             }
355             long endTime = System.currentTimeMillis();
356             logger.info("Current attempt length: " + length + ", Cost time: " + (endTime - startTime) + "ms");
357             if (password != null) {
358                 break;
359             }
360         }
361         JOptionPane.showMessageDialog(this, Objects.requireNonNullElse(password, "Can not find password"));
362         onStopByBruteForceCategory();
363     }

365     private void onStartByDictionarySingleThreadCategory() {
366         logger.info("onStartByDictionarySingleThreadCategory");
367         File dictionaryFile = dictionaryFilePanel.getFile();
368         if (!dictionaryFile.isFile()) {
369             JOptionPane.showMessageDialog(this, "Invalid dictionary file");
370             return;
371         }
372         String charsetName = EncoderDetector.judgeFile(dictionaryFile.getAbsolutePath());
373         logger.info("dictionary file: " + dictionaryFile.getAbsolutePath() + ", charset: " + charsetName);
374         if (charsetName == null) {
375             return;
376         }
377         try (BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(dictionaryFile), charsetName))) {
378             setIsRecovering(true);
379             setProgressMaxValue(Utils.getFileLineCount(dictionaryFile));
380             setProgressBarValue(0);
381             Predicate<String> isRecoveringPredicate = password -> isRecovering;
382             Function<String, Stream<String>> generator = password -> {
383                 increaseProgressBarValue();
384                 return Stream.of(password.toLowerCase(), password.toUpperCase());
385             };
386             Predicate<String> verifier = currentFileChecker::checkPassword;
387             Optional<String> password = br.lines().takeWhile(isRecoveringPredicate).flatMap(generator).filter(verifier).findFirst();
388             if (password.isPresent()) {
389                 JOptionPane.showMessageDialog(this, password.get());
390             } else {
391                 JOptionPane.showMessageDialog(this, "Can not find password");
392             }
393         } catch (FileNotFoundException e) {
394             logger.info("FileNotFoundException");
395         } catch (IOException e) {
396             logger.info("IOException");
397         }
398         onStopByDictionarySingleThreadCategory();
399     }

401     private void onStartByDictionaryMultiThreadCategory() {
402         logger.info("onStartByDictionaryMultiThreadCategory");
403         File dictionaryFile = dictionaryFilePanel.getFile();
404         if (!dictionaryFile.isFile()) {
405             JOptionPane.showMessageDialog(this, "Invalid dictionary file");
406             return;
407         }
408         int threadNum = (Integer) threadNumSpinner.getValue();
409         setIsRecovering(true);
410         setProgressMaxValue(Utils.getFileLineCount(dictionaryFile));
411         setProgressBarValue(0);
412         fileHandle = new FileHandle(currentFileChecker, new AtomicBoolean(false), this);
413         String charsetName = EncoderDetector.judgeFile(dictionaryFile.getAbsolutePath());
414         BigFileReader.Builder builder = new BigFileReader.Builder(dictionaryFile.getAbsolutePath(), fileHandle);
415         bigFileReader = builder.withThreadSize(threadNum).withCharset(charsetName).withBufferSize(1024 * 1024).build();
416         bigFileReader.setCompleteCallback(() -> {
417             if (fileHandle != null && !fileHandle.getSuccess().get()) {
418                 logger.error("Can not find password");
419                 JOptionPane.showMessageDialog(RecoveryPanel.this, "Can not find password");
420                 onStopByDictionaryMultiThreadCategory();
421             }
422         });
423         bigFileReader.start();
424     }

426     private void onStopByBruteForceCategory() {
427         logger.info("onStopByBruteForceCategory");
428         if (workerPool != null && !workerPool.isShutdown()) {
429             workerPool.shutdownNow();
430         }
431         while (workerPool != null && !workerPool.isTerminated()) {
432             try {
433                 final boolean isTimeout = workerPool.awaitTermination(100, TimeUnit.SECONDS);
434                 logger.info("awaitTermination isTimeout: " + isTimeout);
435             } catch (InterruptedException e) {
436                 logger.error("awaitTermination failed");
437                 Thread.currentThread().interrupt();
438             }
439         }
440         setProgressBarValue(0);
441         setIsRecovering(false);
442     }

450     private void onStopByDictionaryMultiThreadCategory() {
451         logger.info("onStopByDictionaryMultiThreadCategory");
452         if (bigFileReader != null) {
453             fileHandle.stop();
454         }
455         if (bigFileReader != null) {
456             bigFileReader.shutdown();
457         }
458         setProgressBarValue(0);
459         setIsRecovering(false);
460     }

462     private int getThreadCount(int charSetSize, int length) {
463         int result = 1;
464         for (int i = 1; i <= length; i++) {
465             result *= charSetSize;
466         }
467         return result;
468     }
