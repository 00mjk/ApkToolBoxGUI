163     public String getComparedKey(File file) {
164         StringBuilder sb = new StringBuilder();
165         if (isSizeChecked.isSelected()) {
166             sb.append("[Size][");
167             sb.append(DigestUtils.md5Hex(String.valueOf(file.length())));
168             sb.append("]");
169         }
170         if (isFileNameChecked.isSelected()) {
171             sb.append("[Filename][");
172             sb.append(DigestUtils.md5Hex(file.getName()));
173             sb.append("]");
174         }
175         if (isMD5Checked.isSelected()) {
176             sb.append("[MD5][");
177             try (InputStream is = new FileInputStream(file)) {
178                 sb.append(DigestUtils.md5Hex(is));
179             } catch (FileNotFoundException e) {
180                 logger.error("getComparedKey FileNotFoundException");
181             } catch (IOException e) {
182                 logger.error("getComparedKey IOException");
183             }
184             sb.append("]");
185         }
186         if (isModifiedTimeChecked.isSelected()) {
187             sb.append("[ModifiedTime][");
188             sb.append(DigestUtils.md5Hex(String.valueOf(file.lastModified())));
189             sb.append("]");
190         }
191         logger.info("path: " + file.getAbsolutePath() + ", key: " + sb);
192         return sb.toString();
193     }

195     class MyMouseListener extends MouseAdapter {
196         @Override
197         public void mouseReleased(MouseEvent e) {
198             super.mouseReleased(e);
199             int r = resultTable.rowAtPoint(e.getPoint());
200             if (r >= 0 && r < resultTable.getRowCount()) {
201                 resultTable.setRowSelectionInterval(r, r);
202             } else {
203                 resultTable.clearSelection();
204             }
205             int rowIndex = resultTable.getSelectedRow();
206             if (rowIndex < 0) {
207                 return;
208             }
209             if (e.isPopupTrigger() && e.getComponent() instanceof JTable) {
210                 JPopupMenu popupmenu = new JPopupMenu();
211                 MyMenuActionListener menuActionListener = new MyMenuActionListener();

213                 openDirMenuItem = new JMenuItem("Open parent folder of this file");

234     class MyMenuActionListener implements ActionListener {
235         @Override
236         public void actionPerformed(ActionEvent actionEvent) {
237             Object source = actionEvent.getSource();
238             if (source.equals(openDirMenuItem)) {
239                 int rowIndex = resultTable.getSelectedRow();
240                 String parentPath = resultTableModel.getValueAt(rowIndex, resultTable.getColumn(DuplicateFilesConstants.COLUMN_NAME_FILE_PARENT).getModelIndex()).toString();
241                 File parent = new File(parentPath);
242                 if (parent.isDirectory()) {
243                     try {
244                         Desktop.getDesktop().open(parent);
245                     } catch (IOException e) {
246                         logger.error("open parent failed: " + parent.getPath());
247                     }
248                 }
249             } else if (source.equals(deleteFileMenuItem)) {
250                 int rowIndex = resultTable.getSelectedRow();
251                 String parentPath = resultTableModel.getValueAt(rowIndex, resultTable.getColumn(DuplicateFilesConstants.COLUMN_NAME_FILE_PARENT).getModelIndex()).toString();
252                 String name = resultTableModel.getValueAt(rowIndex, resultTable.getColumn(DuplicateFilesConstants.COLUMN_NAME_FILE_NAME).getModelIndex()).toString();
253                 File selectedFile = new File(parentPath, name);
254                 String key = getComparedKey(selectedFile);
255                 List<File> files = duplicateFileGroupMap.get(key);
256                 for (File file : files) {
257                     if (!selectedFile.equals(file)) {
258                         continue;
259                     }
260                     files.remove(file);
261                     boolean isSuccessful = file.delete();
262                     logger.info("delete file: " + file.getAbsolutePath() + ", result: " + isSuccessful);
263                     break;
264                 }
265                 resultTableModel.setRowCount(0);
266                 showResult();
267             } else if (source.equals(deleteFilesInSameDirMenuItem)) {
268                 int rowIndex = resultTable.getSelectedRow();
269                 String parentPath = resultTableModel.getValueAt(rowIndex, resultTable.getColumn(DuplicateFilesConstants.COLUMN_NAME_FILE_PARENT).getModelIndex()).toString();
270                 for (Map.Entry<String, List<File>> entry : duplicateFileGroupMap.entrySet()) {
271                     List<File> duplicateFileGroup = entry.getValue();
272                     for (File duplicateFile : duplicateFileGroup) {
273                         String parentPathTmp = duplicateFile.getParent();
274                         if (Objects.equals(parentPath, parentPathTmp)) {
275                             duplicateFileGroup.remove(duplicateFile);
276                             boolean isSuccessful = duplicateFile.delete();
277                             logger.info("delete file: " + duplicateFile.getAbsolutePath() + ", result: " + isSuccessful);
278                             break;
279                         }
280                     }
281                 }
282                 resultTableModel.setRowCount(0);
283                 showResult();
284             } else if (source.equals(deleteFilesInSameDirRecursiveMenuItem)) {
285                 int rowIndex = resultTable.getSelectedRow();
286                 String parentPath = resultTableModel.getValueAt(rowIndex, resultTable.getColumn(DuplicateFilesConstants.COLUMN_NAME_FILE_PARENT).getModelIndex()).toString();
287                 for (Map.Entry<String, List<File>> entry : duplicateFileGroupMap.entrySet()) {
288                     List<File> duplicateFileGroup = entry.getValue();
289                     for (File duplicateFile : duplicateFileGroup) {
290                         String parentPathTmp = duplicateFile.getParent();
291                         if (Objects.equals(parentPath, parentPathTmp) || FilenameUtils.directoryContains(parentPath, parentPathTmp)) {
292                             duplicateFileGroup.remove(duplicateFile);
293                             boolean isSuccessful = duplicateFile.delete();
294                             logger.info("delete file: " + duplicateFile.getAbsolutePath() + ", result: " + isSuccessful);
295                             break;
296                         }
297                     }
298                 }
299                 resultTableModel.setRowCount(0);
300                 showResult();
301             } else {
302                 logger.error("invalid source");
303             }
304         }
305     }

307     class OperationButtonActionListener implements ActionListener {
308         @Override
309         public void actionPerformed(ActionEvent e) {
310             Object source = e.getSource();
311             if (source.equals(searchButton)) {
312                 String[] extensions = null;
313                 if (StringUtils.isNotEmpty(suffixTextField.getText())) {
314                     extensions = suffixTextField.getText().split(",");
315                 }
316                 searchThread = new SearchThread(extensions, isRecursiveSearched.isSelected(), isHiddenFileSearched.isSelected(), duplicateFileGroupMap);
317                 searchThread.start();
318             } else if (source.equals(cancelButton)) {
319                 if (searchThread.isAlive()) {
320                     searchThread.interrupt();
321                 }
322             }
323 
324         }
325     }

327     private void showResult() {
328         SwingUtilities.invokeLater(() -> {
329             int groupIndex = 0;
330             for (Map.Entry<String, List<File>> entry : duplicateFileGroupMap.entrySet()) {
331                 List<File> duplicateFileGroup = entry.getValue();
332                 if (duplicateFileGroup.size() < 2) {
333                     continue;
334                 }
335                 groupIndex++;
336                 for (File duplicateFile : duplicateFileGroup) {
337                     Vector<Object> rowData = getRowVector(groupIndex, duplicateFile);
338                     resultTableModel.addRow(rowData);
339                 }
340             }
341             tabbedPane.setSelectedIndex(1);
342         });
343     }

345     private Vector<Object> getRowVector(int groupIndex, File file) {
346         Vector<Object> rowData = new Vector<>();
347         rowData.add(groupIndex);
348         rowData.add(file.getParent());
349         rowData.add(file.getName());
350         rowData.add(FilenameUtils.getExtension(file.getName()));
351         rowData.add(FileUtilsEx.byteCountToDisplaySize(file.length()));
352         rowData.add(DateTransform.milliSecondToDate(String.valueOf(file.lastModified())));
353         return rowData;
354     }

356     class SearchThread extends Thread {
357         private String[] extensions;
358         private boolean isRecursiveSearched;
359         private boolean isHiddenFileSearched;
360         private Map<String, List<File>> duplicateFileGroupMap;
361 
362         public SearchThread(String[] extensions, boolean isRecursiveSearched, boolean isHiddenFileSearched, Map<String, List<File>> duplicateFileGroupMap) {
363             super();
364             this.extensions = extensions;
365             this.isRecursiveSearched = isRecursiveSearched;
366             this.isHiddenFileSearched = isHiddenFileSearched;
367             this.duplicateFileGroupMap = duplicateFileGroupMap;
368         }
369 
370         @Override
371         public void run() {
372             super.run();
373             duplicateFileGroupMap.clear();
374             List<File> fileList = fileListPanel.getFileList();
375 
376             Set<File> fileSet = new TreeSet<>(fileList);
377             for (File file : fileList) {
378                 fileSet.addAll(FileUtils.listFiles(file, extensions, isRecursiveSearched));
379             }
380 
381             for (File file : fileSet) {
382                 if (Thread.currentThread().isInterrupted()) {
383                     break;
384                 }
385                 if (file.isHidden() && !isHiddenFileSearched) {
386                     continue;
387                 }
388                 String hash = getComparedKey(file);
389                 List<File> list = duplicateFileGroupMap.computeIfAbsent(hash, k -> new LinkedList<>());
390                 list.add(file);
391             }
392             showResult();
393         }
