20  public class FilePanel extends JPanel {
21      private static final Logger LOGGER = LogManager.getLogger(FilePanel.class);
22      private JTextField fileTextField;
23      private JButton chooseButton;
24  
25      private String buttonText;
26  
27      private String currentDirectoryPath;
28  
29      private boolean acceptAllFileFilterUsed;
30  
31      private int fileSelectionMode;
32  
33      private boolean multiSelectionEnabled;
34  
35      private String description;
36  
37      private String[] extensions;
38  
39      private JFileChooser fileChooser;
40  
41      public FilePanel(String buttonText, String currentDirectoryPath, boolean acceptAllFileFilterUsed, int fileSelectionMode,
42                       boolean multiSelectionEnabled, String description, String[] extensions) {
43          super();
44          fileChooser = new JFileChooser();
45          this.buttonText = buttonText;
46          this.currentDirectoryPath = currentDirectoryPath;
47          this.acceptAllFileFilterUsed = acceptAllFileFilterUsed;
48          this.fileSelectionMode = fileSelectionMode;
49          this.multiSelectionEnabled = multiSelectionEnabled;
50          this.description = description;
51          this.extensions = extensions;
52          initUI();
53      }
54  
55      private void initUI() {
56          setLayout(new BoxLayout(this, BoxLayout.X_AXIS));
57  
58          fileTextField = new JTextField();
59          fileTextField.setPreferredSize(new Dimension(600, 0));
60          fileTextField.setTransferHandler(new FileTransferHandler());
61  
62          chooseButton = new JButton(buttonText);
63          chooseButton.addActionListener(new OpenDictionaryFileActionListener());
64  
65          add(fileTextField);
66          add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
67          add(chooseButton);
68      }
69  
70      public File getFile() {
71          File file = new File(fileTextField.getText());
72          return file;
73      }
74  
75      public void updateDescriptionAndFileExtensions(String description, String[] extensions) {
76          this.description = description;
77          this.extensions = extensions;
78          FileNameExtensionFilter filter = new FileNameExtensionFilter(description, extensions);
79          fileChooser.addChoosableFileFilter(filter);
80      }
81  
82      class OpenDictionaryFileActionListener implements ActionListener {
83          @Override
84          public void actionPerformed(ActionEvent e) {
85              fileChooser.setCurrentDirectory(new File(currentDirectoryPath));
86              fileChooser.setAcceptAllFileFilterUsed(acceptAllFileFilterUsed);
87              fileChooser.setFileSelectionMode(fileSelectionMode);
88              fileChooser.setMultiSelectionEnabled(multiSelectionEnabled);
89              FileNameExtensionFilter filter = new FileNameExtensionFilter(description, extensions);
90              fileChooser.addChoosableFileFilter(filter);
91              int returnVal = fileChooser.showOpenDialog(FilePanel.this);
92              if (returnVal == JFileChooser.APPROVE_OPTION) {
93                  File selectedFile = fileChooser.getSelectedFile();
94                  if (selectedFile != null) {
95                      fileTextField.setText(Utils.getCanonicalPathQuiet(selectedFile));
96                  }
97              }
98          }
99      }
100 
101     class FileTransferHandler extends TransferHandler {
102         public FileTransferHandler() {
103         }
104 
105         @Override
106         public boolean importData(JComponent comp, Transferable t) {
107             try {
108                 Object object = t.getTransferData(DataFlavor.javaFileListFlavor);
109                 for (File file : (List<File>)object) {
110                     fileTextField.setText(file.getCanonicalPath());
111                 }
112                 return true;
113             } catch (IOException e) {
114                 LOGGER.error("importData failed: IOException");
115             } catch (UnsupportedFlavorException e) {
116                 LOGGER.error("importData failed: UnsupportedFlavorException");
117             }
118             return false;
119         }
120 
121         @Override
122         public boolean canImport(JComponent jComponent, DataFlavor[] dataFlavors) {
123             for (DataFlavor dataFlavor : dataFlavors) {
124                 if (DataFlavor.javaFileListFlavor.equals(dataFlavor)) {
125                     return true;
126                 }
127             }
128             return false;
129         }
130     }
