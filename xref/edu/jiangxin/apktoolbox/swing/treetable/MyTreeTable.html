7   public class MyTreeTable extends JTable {

9       private MyTreeTableCellRenderer tree;

12      public MyTreeTable(MyAbstractTreeTableModel treeTableModel) {
13          super();
14  
15          // JTree erstellen.
16          tree = new MyTreeTableCellRenderer(this, treeTableModel);
17  
18          // Modell setzen.
19          super.setModel(new MyTreeTableModelAdapter(treeTableModel, tree));
20  
21          // Gleichzeitiges Selektieren fuer Tree und Table.
22          MyTreeTableSelectionModel selectionModel = new MyTreeTableSelectionModel();
23          tree.setSelectionModel(selectionModel); //For the tree
24          setSelectionModel(selectionModel.getListSelectionModel()); //For the table
25  
26  
27          // Renderer fuer den Tree.
28          setDefaultRenderer(MyTreeTableModel.class, tree);
29          // Editor fuer die TreeTable
30          setDefaultEditor(MyTreeTableModel.class, new MyTreeTableCellEditor(tree, this));
31  
32          // Kein Grid anzeigen.
33          setShowGrid(false);
34  
35          // Keine Abstaende.
36          setIntercellSpacing(new Dimension(0, 0));
37  
38      }
