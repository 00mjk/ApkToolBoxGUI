182     private void initData() {
183         BufferedReader bfReader = null;
184         bfReader = new BufferedReader(new StringReader(Constants.data));
185         String tempString;
186         try {
187             tempString = bfReader.readLine();
188             while (tempString.length() > 3) {
189                 dataHashtable.put(StringUtils.substringBetween(tempString, "[", "]"), StringUtils.substringBetween(tempString, "{", "}"));
190                 tempString = bfReader.readLine();
191             }
192         } catch (IOException e) {
193             JOptionPane.showMessageDialog(null, "数据载入失败!");
194         } catch (NullPointerException e) {
195             System.out.println("数据载入完成->共加载数据" + dataHashtable.size() + "条");
196         }
197 
198     }

202 class ButtonListener implements ActionListener {
203     public void actionPerformed(ActionEvent e) {
204         String which = ((JButton) e.getSource()).getText();
205         if (which.equals("AC")) {
206             RelationShipConvertPanel.now = "你";
207             RelationShipConvertPanel.isOlder = true;
208             RelationShipConvertPanel.olderButton.setBackground(new Color(217, 2, 0));
209             RelationShipConvertPanel.inpuTextArea.setText("");
210             RelationShipConvertPanel.resultTextArea.setText("");
211         } else if (which.equals("=")) {
212             if (RelationShipConvertPanel.now.charAt(0) == '<') {
213                 String[] oldChooseString = StringUtils.substringsBetween(RelationShipConvertPanel.now, "<", ">");
214                 RelationShipConvertPanel.now = RelationShipConvertPanel.isOlder ? oldChooseString[0] : oldChooseString[1];
215                 RelationShipConvertPanel.inpuTextArea.setText(RelationShipConvertPanel.inpuTextArea.getText() + (RelationShipConvertPanel.isOlder ? "(比你年长)" : "(比你年轻)"));
216             }
217             RelationShipConvertPanel.olderButton.setBackground(new Color(217, 2, 0));
218             RelationShipConvertPanel.isOlder = true;
219             RelationShipConvertPanel.resultTextArea.setText(RelationShipConvertPanel.now);
220             //RelationShipConvertPanel.now="你";
221         } else if (which.equals("小")) {
222             RelationShipConvertPanel.isOlder = false;
223             RelationShipConvertPanel.olderButton.setBackground(new Color(217, 2, 0));
224         } else if (which.equals("←")) {
225 
226         } else if (which.equals("的")) {
227             String temp = "";
228             RelationShipConvertPanel.olderButton.setBackground(new Color(217, 2, 0));
229             if (RelationShipConvertPanel.now.charAt(0) == '<') {
230                 String[] oldChooseString = StringUtils.substringsBetween(RelationShipConvertPanel.now, "<", ">");
231                 RelationShipConvertPanel.now = RelationShipConvertPanel.isOlder ? oldChooseString[0] : oldChooseString[1];
232                 temp = RelationShipConvertPanel.isOlder ? "(比你年长)" : "(比你年轻)";
233             }
234             RelationShipConvertPanel.inpuTextArea.setText(RelationShipConvertPanel.inpuTextArea.getText() + temp + which);
235             RelationShipConvertPanel.isOlder = true;
236         } else {//按下关系按钮时
237             RelationShipConvertPanel.olderButton.setBackground(new Color(217, 2, 0));
238             RelationShipConvertPanel.inpuTextArea.setText(RelationShipConvertPanel.inpuTextArea.getText() + which);
239             RelationShipConvertPanel.now = getNext(which).substring(2);
240             if (RelationShipConvertPanel.now.charAt(0) == '<') {
241                 RelationShipConvertPanel.olderButton.setBackground(new Color(212, 98, 2));
242             }
243             RelationShipConvertPanel.isOlder = true;
244         }
245 
246     }

248     String getNext(String what) {
249         String relationsString = RelationShipConvertPanel.dataHashtable.get(RelationShipConvertPanel.now);
250         String[] relation = null;
251         if (relationsString != null) {
252             relation = StringUtils.substringsBetween(relationsString, "(", ")");
253         } else {
254             RelationShipConvertPanel.resultTextArea.setText("超出计算范围");
255             return RelationShipConvertPanel.now;
256         }
257         if (what.equals("父")) {
258             return relation[0];
259         } else if (what.equals("母")) {
260             return relation[1];
261         } else if (what.equals("兄")) {
262             return relation[2];
263         } else if (what.equals("弟")) {
264             return relation[3];
265         } else if (what.equals("姐")) {
266             return relation[4];
267         } else if (what.equals("妹")) {
268             return relation[5];
269         } else if (what.equals("夫")) {
270             return relation[6];
271         } else if (what.equals("妻")) {
272             return relation[7];
273         } else if (what.equals("儿")) {
274             return relation[8];
275         } else if (what.equals("女")) {
276             return relation[9];
277         } else {
278             return RelationShipConvertPanel.now;
279         }
