203     public void createOperationPanel() {
204         operationPanel = new JPanel();
205         operationPanel.setLayout(new BoxLayout(operationPanel, BoxLayout.X_AXIS));
206 
207         JButton rgb2OthersConvertBtn = new JButton("RGB->Others");
208         rgb2OthersConvertBtn.addActionListener(e -> {
209             int red = (Integer) redSpinner.getValue();
210             int green = (Integer) greenSpinner.getValue();
211             int blue = (Integer) blueSpinner.getValue();
212             color = new Color(red, green, blue);
213             syncToOthersFormat();
214         });
215 
216         JButton hex2OthersConvertBtn = new JButton("Hex->Others");
217         hex2OthersConvertBtn.addActionListener(e -> {
218             String hex = hexTextField.getText();
219             color = ColorUtils.hex2Color(hex);
220             syncToOthersFormat();
221         });
222 
223         JButton hsb2OthersConvertBtn = new JButton("HSB->Others");
224         hsb2OthersConvertBtn.addActionListener(e -> {
225             float hue = Float.valueOf(hueTextField.getText());
226             float saturation = Float.valueOf(saturationTextField.getText());
227             float brightness = Float.valueOf(brightnessTextField.getText());
228             color = Color.getHSBColor(hue, saturation, brightness);
229             syncToOthersFormat();
230         });
231 
232         JButton cmyk2OthersConvertBtn = new JButton("HSB->Others");
233         cmyk2OthersConvertBtn.addActionListener(e -> {
234             float cyan = Float.valueOf(cyanTextField.getText());
235             float magenta = Float.valueOf(magentaTextField.getText());
236             float yellow = Float.valueOf(yellowTextField.getText());
237             float black = Float.valueOf(blackTextField.getText());
238             color = ColorUtils.cmyk2Color(cyan, magenta, yellow, black);
239             syncToOthersFormat();
240         });
241 
242         colorBoxTextField = new JTextField();
243         colorBoxTextField.setEditable(false);
244         colorBoxTextField.setOpaque(false);
245 
246         operationPanel.add(rgb2OthersConvertBtn);
247         operationPanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
248         operationPanel.add(hex2OthersConvertBtn);
249         operationPanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
250         operationPanel.add(hsb2OthersConvertBtn);
251         operationPanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
252         operationPanel.add(cmyk2OthersConvertBtn);
253         operationPanel.add(Box.createHorizontalStrut(Constants.DEFAULT_X_BORDER));
254         operationPanel.add(colorBoxTextField);
255     }

257     private void createOrdinaryColorTable() {
258         colorTable = new JTable(new ColorDefaultTableModel(ColorTableConstant.COLOR_TABLE_ROW_DATA, ColorTableConstant.COLUMN_NAMES));
259         colorTableScrollPane = new JScrollPane(colorTable);
260         int columnCount = colorTable.getColumnCount();
261         for (int i = 0; i < columnCount; i++) {
262             colorTable.getColumn(colorTable.getColumnName(0)).setCellRenderer(new ColorTableCellRenderer());
263         }
264     }
265 
266     private class ColorDefaultTableModel extends DefaultTableModel {
267         public ColorDefaultTableModel(Object[][] data, Object[] columnNames) {
268             super(data, columnNames);
269         }
270 
271         @Override
272         public boolean isCellEditable(int row, int column) {
273             return true;
274         }
275 
276     }
277 
278     private class ColorTableCellRenderer extends DefaultTableCellRenderer {
279         @Override
280         public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
281             Component renderer = super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, column);
282             renderer.setBackground(ColorUtils.hex2Color((String) colorTable.getValueAt(row, 3)));
283             return renderer;
284         }
285     }

287     private void syncToOthersFormat() {
288         redSpinner.setValue(color.getRed());
289         greenSpinner.setValue(color.getGreen());
290         blueSpinner.setValue(color.getBlue());
291 
292         hexTextField.setText(ColorUtils.color2Hex(color));
293 
294         float[] hsbArray = Color.RGBtoHSB(color.getRed(), color.getGreen(), color.getBlue(), null);
295         hueTextField.setText(new DecimalFormat("0.0000").format(hsbArray[0]));
296         saturationTextField.setText(new DecimalFormat("0.0000").format(hsbArray[1]));
297         brightnessTextField.setText(new DecimalFormat("0.0000").format(hsbArray[2]));
298 
299         float[] cmykArray = ColorUtils.color2Cmyk(color);
300         cyanTextField.setText(new DecimalFormat("0.0000").format(cmykArray[0]));
301         magentaTextField.setText(new DecimalFormat("0.0000").format(cmykArray[1]));
302         yellowTextField.setText(new DecimalFormat("0.0000").format(cmykArray[2]));
303         blackTextField.setText(new DecimalFormat("0.0000").format(cmykArray[3]));
304 
305         colorBoxTextField.setBackground(color);
306         //paint the color box with the converted output color
307         colorBoxTextField.setOpaque(true);
308     }
