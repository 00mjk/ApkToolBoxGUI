14      public static Color cmyk2Color(float cyan, float magenta, float yellow, float key) {
15          int red = (int) (255 * (1 - cyan) * (1 - key));
16          int green = (int) (255 * (1 - magenta) * (1 - key));
17          int blue = (int) (255 * (1 - yellow) * (1 - key));
18          return new Color(red, green, blue);
19      }

21      public static float[] color2Cmyk(Color color) {
22          float redInFloat = color.getRed() / 255f;
23          float greenInFloat = color.getGreen() / 255f;
24          float blueInFloat = color.getBlue() / 255f;
25          float key = 1 - Math.max(redInFloat, blueInFloat);
26          float cyan = (1 - redInFloat - key) / (1 - key);
27          float magenta = (1 - greenInFloat - key) / (1 - key);
28          float yellow = (1 - blueInFloat - key) / (1 - key);
29          return new float[]{cyan, magenta, yellow, key};
30      }
