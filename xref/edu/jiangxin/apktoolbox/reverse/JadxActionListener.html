25  public class JadxActionListener implements ActionListener {
26  
27      private static final Logger logger = LogManager.getLogger(JadxActionListener.class);
28  
29      private Configuration conf;
30  
31      public JadxActionListener() {
32          conf = Utils.getConfiguration();
33      }
34  
35      @Override
36      public void actionPerformed(ActionEvent event) {
37          String toolPath = conf.getString(Constants.JADX_PATH_KEY);
38          File toolFile = null;
39          if (!StringUtils.isEmpty(toolPath)) {
40              toolFile = new File(toolPath);
41          }
42          if (StringUtils.isEmpty(toolPath) || toolFile == null || !toolFile.exists() || !toolFile.isFile()) {
43              Toolkit.getDefaultToolkit().beep();
44              JOptionPane.showMessageDialog(null, "Need Configuration", "ERROR",
45                      JOptionPane.ERROR_MESSAGE);
46              return;
47          }
48  
49          String cmd = conf.getString(Constants.JADX_PATH_KEY);
50          Utils.unBlockedExecutor(cmd);
51      }
52  }
